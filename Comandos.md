# =======================================
# Cria um novo projeto Expo com o template "blank"
npx create-expo-app@latest nome-projeto --template blank

# ========================================
# Cria o arquivo package.json interativamente.
npm init
# ========================================
# Instala pacotes no projeto e adiciona ao  package.json.
npm install express

# ========================================
# Instala todas as dependências listadas no package.json.
npm install
# ========================================
# Inicia a aplicação (se configurado no package.json).
npm start
